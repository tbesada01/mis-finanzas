create table movements ( id uuid primary key default gen_random_uuid(), created_at timestamptz default now(), type text not null, currency text default 'ARS', desc text not null, amount numeric not null, my_part numeric not null, to_recover numeric default 0, date date not null, category text, method text, notes text, shared text default 'no', shared_with text ); create table goals ( id uuid primary key default gen_random_uuid(), created_at timestamptz default now(), name text not null, target numeric, saved numeric default 0, deadline text, currency text default 'ARS', goal_type text default 'monto', pct numeric ); alter table movements enable row level security; alter table goals enable row level security; create policy "public access movements" on movements for all using (true) with check (true); create policy "public access goals" on goals for all using (true) with check (true);
